00561 template <class T> inline T
00562 Specbnd<T>::_LambdaS
00563 ( const fadbad::F<T> &a, unsigned int n )
00564 {
00565   if( !n ) return 0.;
00566   if( n == 1 ) return a.size()? Op<T>::sqr( a[0] ): 0.;
00567   double upbnd=0.;
00568   for (unsigned int i=0; i<n; i++)
00569     if( a.size() ) upbnd += Op<T>::u( Op<T>::sqr( a[i] ) );
00570   return Op<T>::zeroone() * upbnd;
00571 }

00573 template <class T> inline T
00574 Specbnd<T>::_LambdaT
00575 ( const fadbad::F<T> &a, const fadbad::F<T> &b, unsigned int n )
00576 {
00577   if( !n ) return 0.;
00578   if( n == 1 ) return (a.size() && b.size())? 2.*a[0]*b[0]: 0.;
00579   double upbnda=0., upbndb=0.;
00580   for (unsigned int i=0; i<n; i++){
00581     if( a.size() ) upbnda += Op<T>::u( Op<T>::sqr( a[i] ) );
00582     if( b.size() ) upbndb += Op<T>::u( Op<T>::sqr( b[i] ) );
00583   }
00584   T lamb = 2.*(Op<T>::zeroone()-0.5) * std::sqrt(upbnda*upbndb);
00585   for (unsigned int i=0; i<n; i++)
00586     if( a.size() && b.size() ) lamb += a[i]*b[i];
00587   return lamb;
00588 }

00591 Specbnd<T>::spectrum
00592 ( const fadbad::B< fadbad::F< double > >* D2X )
00593 {
00594   if( !D2X ) throw typename Specbnd<T>::Exceptions( Specbnd<T>::Exceptions::SPECTR );
00595   const unsigned int N = D2X->val().size();
00596   if( !N ) throw typename Specbnd<T>::Exceptions( Specbnd<T>::Exceptions::SPECTR );
00597   double*H = new double[N*N];
00598 
00599   for( unsigned int j=0; j<N; j++ )
00600     for( unsigned int i=j; i<N; i++ )
00601       H[j*N+i] = H[i*N+j] = ( i==j? D2X[i].deriv(0).deriv(i):
00602         0.5*(D2X[i].deriv(0).deriv(j)+D2X[j].deriv(0).deriv(i)) );
00603 #ifdef MC__SPECBND_DEBUG_SPECTRUM
00604   mc::display( N, N, H, N, "\nMatrix H", std::cout );
00605 #endif

00608   delete[] H;
00609   if( !D ) throw typename Specbnd<T>::Exceptions( Specbnd<T>::Exceptions::SPECTR );
00610   std::pair<double,double> spbnd = std::make_pair( min(N,D), max(N,D) );
00611   delete[] D;
00612   return spbnd;

00652 template <typename T> inline std::pair<double,double>
00653 Specbnd<T>::_gershgorin_bound
00654 ( const fadbad::B< fadbad::F< T > >* D2X )
00655 {
00656   const unsigned int N = D2X->val().size();
00657   if( !N ) throw typename Specbnd<T>::Exceptions( Specbnd<T>::Exceptions::HESSBND );
00658 
00659   std::pair<double,double> spbnd;
00660   for( unsigned int i=0; i<N; i++ ){
00661     double ri = 0.;
00662     for( unsigned int j=0; j<N; j++ ){
00663       if( j == i ) continue;
00664       T D2Fij;
00665       if( !inter( D2Fij, D2X[i].deriv(0).deriv(j), D2X[i].deriv(0).deriv(j) ) )
00666         D2Fij = D2X[i].deriv(0).deriv(j); 
00667       ri += Op<T>::abs( D2Fij );
00668     }
00669 
00670     if( !i ){
00671       spbnd.first  = Op<T>::l(D2X[i].deriv(0).deriv(i)) - ri;
00672       spbnd.second = Op<T>::u(D2X[i].deriv(0).deriv(i)) + ri;
00673     }
00674     else{
00675       spbnd.first  = std::min( spbnd.first,  Op<T>::l(D2X[i].deriv(0).deriv(i)) - ri );
00676       spbnd.second = std::max( spbnd.second, Op<T>::u(D2X[i].deriv(0).deriv(i)) + ri );
00677     }
00678   }
00679   return spbnd;
00680 }

00682 template <typename T> inline std::pair<double,double>
00683 Specbnd<T>::_hertzrohn_bound
00684 ( const fadbad::B< fadbad::F< T > >* D2X )
00685 {
00686   const unsigned int N = D2X->val().size();
00687   if( !N ) throw typename Specbnd<T>::Exceptions( Specbnd<T>::Exceptions::HESSBND );
00688 
00689   // Form matrix S^{(N)} recursively
00690   unsigned int col_S = 2;
00691   short *S = new short[col_S];
00692   S[0] = 1; S[1] = -1;
00693 #ifdef MC__SPECBND_DEBUG_HESSBND
00694   mc::display( 1, col_S, S, 1, "\nMatrix S1", std::cout );
00695 #endif
00696   for( unsigned int k=1; k<N; k++, col_S*=2 ){
00697     short *Sprev = new short[k*col_S];
00698     for( unsigned int i=0; i<k; i++ )
00699       for( unsigned int j=0; j<col_S; j++ )
00700         Sprev[j*k+i] = S[j*k+i];
00701     delete[] S; S = new short[(k+1)*2*col_S];
00702     for( unsigned int i=0; i<k; i++ ){
00703       for( unsigned int j=0; j<col_S; j++ )
00704         S[j*(k+1)+i] = S[(col_S+j)*(k+1)+i] = Sprev[j*k+i];
00705     }
00706     for( unsigned int j=0; j<col_S; j++ ){
00707       S[j*(k+1)+k] = 1; S[(col_S+j)*(k+1)+k] = -1;
00708     }
00709     delete[] Sprev;
00710 #ifdef MC__SPECBND_DEBUG_HESSBND
00711     mc::display( k+1, 2*col_S, S, k+1, "\nMatrix Sk", std::cout );
00712 #endif
00713   }
00714 
00715   // Compute lower and upper bound on spectral radius
00716   std::pair<double,double> spbnd;
00717   double *Lk = new double[N*N], *Uk = new double[N*N];
00718   for( unsigned int k=0; k<col_S; k++ ){
00719     
00720     for( unsigned int j=0; j<N; j++ )
00721       for( unsigned int i=j; i<N; i++ ){
00722         if( i == j ){
00723      Lk[i*N+i] = Op<T>::l( D2X[i].deriv(0).deriv(i) );
00724      Uk[i*N+i] = Op<T>::u( D2X[i].deriv(0).deriv(i) );
00725           continue;
00726    }
00727         T D2Fij;
00728         if( !inter( D2Fij, D2X[i].deriv(0).deriv(j), D2X[j].deriv(0).deriv(i) ) )
00729           D2Fij = D2X[i].deriv(0).deriv(j);
00730         Lk[i*N+j] = Lk[j*N+i] = S[k*N+i]*S[k*N+j]==1? Op<T>::l( D2Fij ): Op<T>::u( D2Fij );
00731         Uk[i*N+j] = Uk[j*N+i] = S[k*N+i]*S[k*N+j]==1? Op<T>::u( D2Fij ): Op<T>::l( D2Fij );
00732       }
00733 #ifdef MC__SPECBND_DEBUG_HESSBND
00734     mc::display( N, N, Lk, N, "\nMatrix Lk", std::cout );
00735     mc::display( N, N, Uk, N, "\nMatrix Uk", std::cout );
00736 #endif
00737 
00738     double*DLk = mc::dsyev_wrapper( N, Lk );
00739     if( !DLk ){
00740       delete[] Lk; delete[] Uk; delete[] S;
00741       throw typename Specbnd<T>::Exceptions( Specbnd<T>::Exceptions::HESSBND );
00742     }
00743 #ifdef MC__SPECBND_DEBUG_HESSBND
00744     mc::display( 1, N, DLk, 1, "\nMatrix DLk", std::cout );
00745 #endif
00746     spbnd.first = k? std::min( spbnd.first, min(N,DLk) ): min(N,DLk);
00747     delete[] DLk;
00748 
00749     double*DUk = mc::dsyev_wrapper( N, Uk );
00750     if( !DUk ){
00751       delete[] Lk; delete[] Uk; delete[] S;
00752       throw typename Specbnd<T>::Exceptions( Specbnd<T>::Exceptions::HESSBND );
00753     }
00754 #ifdef MC__SPECBND_DEBUG_HESSBND
00755     mc::display( 1, N, DUk, 1, "\nMatrix DUk", std::cout );
00756 #endif
00757     spbnd.second = k? std::max( spbnd.second, max(N,DUk) ): max(N,DUk);
00758     delete[] DUk;
00759 
00760   }
00761   delete[] Lk; delete[] Uk; delete[] S;
00762   return spbnd;
00763 }

00834     delete[] Sprev;
00835 #ifdef MC__SPECBND_DEBUG_HESSBND
00836     mc::display( k+1, 2*col_S, S, k+1, "\nMatrix Sk", std::cout );
00837 #endif
00838   }
00839 
00840 #ifdef MC__SPECBND_DEBUG_HESSBND
00841   T *H = new T[N*N];
00842   for( unsigned int j=0; j<N; j++ )
00843     for( unsigned int i=j; i<N; i++ ){
00844       if( i == j ){
00845         H[i*N+i] = D2F.deriv(i).deriv(i);
00846         continue;
00847       }
00848       T D2Fij;
00849       if( !inter( D2Fij, D2F.deriv(i).deriv(j), D2F.deriv(j).deriv(i) ) )
00850         D2Fij = D2F.deriv(i).deriv(j);
00851       H[i*N+j] = H[j*N+i] = D2Fij;
00852     }
00853   mc::display( N, N, H, N, "\nMatrix H", std::cout );
00854   delete[] H;
00855 #endif
00856 
00857   // Compute lower and upper bound on spectral radius
00858   std::pair<double,double> spbnd;
00859   double *Lk = new double[N*N], *Uk = new double[N*N];
00860   for( unsigned int k=0; k<col_S; k++ ){
00861     
00862     for( unsigned int j=0; j<N; j++ )
00863       for( unsigned int i=j; i<N; i++ ){
00864         if( i == j ){
00865      Lk[i*N+i] = Op<T>::l( D2F.deriv(i).deriv(i) );
00866      Uk[i*N+i] = Op<T>::u( D2F.deriv(i).deriv(i) );
00867           continue;
00868    }
00869         T D2Fij;
00870         if( !inter( D2Fij, D2F.deriv(i).deriv(j), D2F.deriv(j).deriv(i) ) )
00871           D2Fij = D2F.deriv(i).deriv(j);
00872         Lk[i*N+j] = Lk[j*N+i] = S[k*N+i]*S[k*N+j]==1? Op<T>::l( D2Fij ): Op<T>::u( D2Fij );
00873         Uk[i*N+j] = Uk[j*N+i] = S[k*N+i]*S[k*N+j]==1? Op<T>::u( D2Fij ): Op<T>::l( D2Fij );
00874       }
00875 #ifdef MC__SPECBND_DEBUG_HESSBND
00876     mc::display( N, N, Lk, N, "\nMatrix Lk", std::cout );
00877     mc::display( N, N, Uk, N, "\nMatrix Uk", std::cout );
00878 #endif
00879 
00880     double*DLk = mc::dsyev_wrapper( N, Lk );
00881     if( !DLk ){
00882       delete[] Lk; delete[] Uk; delete[] S;
00883       throw typename Specbnd<T>::Exceptions( Specbnd<T>::Exceptions::HESSBND );
00884     }
00885 #ifdef MC__SPECBND_DEBUG_HESSBND
00886     mc::display( 1, N, DLk, 1, "\nMatrix DLk", std::cout );
00887 #endif
00888     spbnd.first = k? std::min( spbnd.first, min(N,DLk) ): min(N,DLk);
00889     delete[] DLk;
00890 
00891     double*DUk = mc::dsyev_wrapper( N, Uk );
00892     if( !DUk ){
00893       delete[] Lk; delete[] Uk; delete[] S;
00894       throw typename Specbnd<T>::Exceptions( Specbnd<T>::Exceptions::HESSBND );
00895     }
00896 #ifdef MC__SPECBND_DEBUG_HESSBND
00897     mc::display( 1, N, DUk, 1, "\nMatrix DUk", std::cout );
00898 #endif
00899     spbnd.second = k? std::max( spbnd.second, max(N,DUk) ): max(N,DUk);
00900     delete[] DUk;
00901 
00902   }
00903   delete[] Lk; delete[] Uk; delete[] S;
00904   return spbnd;
00905 }
