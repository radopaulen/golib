00710   std::pair<double,DOSTRUCT::t_CTR> g;
00711   maxinfeas = 0.;
00712   for( unsigned int ic=0; ic<_nc; ic++ ){
00713     g = _pb.CTR( ic, p, xk, ns );

00715     case DOSTRUCT::EQ:

00717     case DOSTRUCT::LE:
00718       maxinfeas = std::max(maxinfeas,g.first); break;
00719     case DOSTRUCT::GE:
00720       maxinfeas = std::max(maxinfeas,-g.first); break;
00721     }
00722   }
00723 
00724   // Compute objective function value
00725   f = _pb.OBJ( p, xk, ns ).first;
00726 
00727   // Clean-up
00728   for( unsigned int is=0; is<=ns; is++ ) delete[] xk[is];
00729   delete[] xk;

01432     delete[] xk; delete[] xkp; delete[] lkp; delete[] qp;

01434   }
01435 
01436   // Intermediate variables
01437   t_F **Fxk = new t_F*[ns+1];
01438   t_B **Bxk = new t_B*[ns+1];
01439   t_BF **BFxk = new t_BF*[ns+1];

01441     Fxk[is] = new t_F[_nx];
01442     Bxk[is] = new t_B[_nx];
01443     BFxk[is] = new t_BF[_nx];
01444   }
01445 
01446   for( unsigned int jp=0; jp<_np; jp++ ){
01447 
01448     // Initialize variables for forward-backward AD
01449     for( unsigned int ip=0; ip<_np; ip++ ){
01450       _Fp[ip] = p[ip];
01451       _Fp[ip].diff(0,1) = ( ip==jp? 1.:0. );

01453       _BFp[ip] = _Fp[ip];

01456       for( unsigned int ix=0; ix<_nx; ix++ ){

01459         Bxk[is][ix] = xk[is][ix];
01460         BFxk[is][ix] = Fxk[is][ix];
01461       }
01462     }
01463 
01464     // Compute objective and constraint function values and first derivatives

01466     f = OBJ.x();
01467     fp[jp] = OBJ.d(0);
01468     for( unsigned int ic=0; ic<_nc; ic++ ){
01469       t_F CTR = _pb.CTR( ic, _Fp, Fxk, ns ).first;

01488     DIC.diff(1,2);

01497       }
01498       break;
01499     case Options::HESSLOWER:

01509       }

01512       for( unsigned int ip=0; ip<=jp; ip++ ){

01515      + qp[ip+jp*_np] + _BFp[jp].d(0).d(0) + _BFp[jp].d(1).d(0)
