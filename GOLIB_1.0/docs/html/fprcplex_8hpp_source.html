00247 template <typename T> inline void
00248 FPRCplex<T>::_append_cut
00249 ( const FPCut<T>* pCut )
00250 {
00251   bool isSOS = ( pCut->type() == FPCut<T>::SOS1
00252               || pCut->type() == FPCut<T>::SOS2 );
00253   IloNumVarArray VarSOS( *_env, (isSOS? pCut->nvar(): 0) );
00254   IloNumArray WeiSOS( *_env, (isSOS? pCut->nvar(): 0) );
00255 
00256   IloExpr lhs( *_env );
00257   for( unsigned int k=0; k<pCut->nvar(); k++ ){
00258     switch( pCut->idvar()[k].first ){
00259       case FPVar<T>::VARCONT: case FPVar<T>::VARBIN:
00260       { it_IloNumVar ivar = _var.find( pCut->idvar()[k].second );
00261         if( ivar==_var.end() )
00262           throw typename FPRelax<T>::Exceptions( FPRelax<T>::Exceptions::INTERNAL );
00263         lhs += pCut->coef()[k] * (*ivar).second;
00264         if( isSOS ){ VarSOS[k] = (*ivar).second;
00265                      WeiSOS[k] = (double)k/(double)pCut->nvar(); };
00266         break; }
00267       default:
00268       { it_IloNumVar iaux = _aux.find( pCut->idvar()[k].second );
00269         if( iaux==_aux.end() )
00270           throw typename FPRelax<T>::Exceptions( FPRelax<T>::Exceptions::INTERNAL );
00271         lhs += pCut->coef()[k] * (*iaux).second;
00272         if( isSOS ){ VarSOS[k] = (*iaux).second;
00273                      WeiSOS[k] = (double)k/(double)pCut->nvar(); };
00274         break; }
00275     }
00276   }
00277   IloRange constr;
00278   switch( pCut->type() ){
00279     case FPCut<T>::SOS1:
00280       _model->add( IloSOS1( *_env, VarSOS, WeiSOS ) );
00281       constr = ( lhs == pCut->rhs() ); break;
00282     case FPCut<T>::SOS2:
00283       _model->add( IloSOS2( *_env, VarSOS, WeiSOS ) );
00284       constr = ( lhs == pCut->rhs() ); break;
00285     case FPCut<T>::EQ:
00286       constr = ( lhs == pCut->rhs() ); break;
00287     case FPCut<T>::LE:
00288       constr = ( lhs <= pCut->rhs() ); break;
00289     case FPCut<T>::GE:
00290       constr = ( lhs >= pCut->rhs() ); break;
00291   }
00292   _model->add( constr );
00293   _cut.insert( std::make_pair( pCut->op(), constr ) );
00294 }

00398 template <typename T> inline void
00399 FPRCplex<T>::set_objective

00401 {
00402   // Declare objective in LP model
00403   FPRelax<T>::set_objective( type, objVar );
00404 
00405   IloNum vobj = 0;
00406   it_IloNumVar iobjVar;
00407   switch( objVar.id().first ){
00408     case FPVar<T>::VARCONT: case FPVar<T>::VARBIN:
00409       iobjVar = _var.find( objVar.id().second );
00410       if( iobjVar == _var.end() )
00411         throw typename FPRelax<T>::Exceptions( FPRelax<T>::Exceptions::INTERNAL );
00412       break;
00413     default:
00414       iobjVar = _aux.find( objVar.id().second );
00415       if( iobjVar == _aux.end() )
00416         vobj = ( objVar.num().t==FPVarNum<T>::INT? objVar.num().n: objVar.num().x );
00417       break;
00418   }
00419   
00420   if( _setObj ) _model->remove( _curObj );
00421   switch( type ){
00422     case FPRCplex<T>::MIN:
00423       if( iobjVar == _aux.end() )
00424         _curObj = IloMinimize( *_env, vobj );
00425       else
00426         _curObj = IloMinimize( *_env, (*iobjVar).second );
00427       break;
00428     case FPRCplex<T>::MAX:
00429       if( iobjVar == _aux.end() )
00430         _curObj = IloMaximize( *_env, vobj );
00431       else
00432         _curObj = IloMaximize( *_env, (*iobjVar).second );
00433       break;
00434   }
00435   _model->add( _curObj );
00436   _setObj = true;
