00481 inline typename SBB<T>::TYPE
00482 DOSBB<T,DO>::_set_DO_structure
00483 ()
00484 {
00485   Structure *pS = new Structure[_np];
00486   for( unsigned int ip=0; ip<_np; ip++ )
00487     pS[ip].indep(ip);   
00488   Structure **xkS = new Structure*[_data.ns+1];
00489 
00490   // Structure of initial conditions in IVP
00491   xkS[0] = new Structure[_nx];
00492   for( unsigned int ix=0; ix<_nx; ix++ ){
00493     xkS[0][ix] = DO().IC( ix, pS );
00494 #ifdef MC__DOSBB_DEBUG
00495     std::cout << "xkS[0][" << ix << "] <- " << xkS[0][ix]
00496               << std::endl;
00497 #endif
00498   }
00499 
00500   // Structure of right-hand side in each stage in IVP
00501   for( unsigned int is=0; is<_data.ns; is++ ){
00502     xkS[is+1] = new Structure[_nx];
00503     for( unsigned int ix=0; ix<_nx; ix++ )
00504       xkS[is+1][ix] = DO().RHS( ix, pS, xkS[is], is ) + xkS[is][ix];
00505     bool iterate = true;
00506     while( iterate ){
00507       iterate = false;
00508       for( unsigned int ix=0; ix<_nx; ix++ ){
00509         Structure xiSnew = DO().RHS( ix, pS, xkS[is+1], is ) + xkS[is+1][ix];
00510         if( xkS[is+1][ix] != xiSnew ){
00511      xkS[is+1][ix] = xiSnew; iterate = true;
00512    }
00513       }
00514     }
00515 #ifdef MC__DOSBB_DEBUG
00516     for( unsigned int ix=0; ix<_nx; ix++ )
00517       std::cout << "xkS[" << is+1 << "][" << ix << "] <- " << xkS[is+1][ix]
00518                 << std::endl;
00519 #endif
00520   }
00521 
00522   // Variable dependence in IVP
00523   _IVPdepend.clear();
00524   for( unsigned int ip=0; ip<_np; ip++ ){
00525     bool keepgoing = true;
00526     for( unsigned int ix=0; keepgoing && ix<_nx; ix++ ){
00527       std::map<int,bool>& IVPdep = xkS[_data.ns][ix].dep();
00528       std::map<int,bool>::iterator it = IVPdep.find(ip);
00529       if( it != IVPdep.end() ){
00530         _IVPdepend.insert(ip);
00531         keepgoing = false;
00532       }
00533     }
00534   }
00535 
00536   // Structure of objective function and constraints (parameter dependence)
00537   std::pair<Structure,DOSTRUCT::t_OBJ> pobj = DO().OBJ( pS, xkS, _data.ns );
00538   typename SBB<T>::TYPE pb;
00539   switch( pobj.second ){
00540     case DOSTRUCT::MIN: pb = SBB<T>::MINIMIZE; break;
00541     default:            pb = SBB<T>::MAXIMIZE; break;
00542   }
00543   _DOstruct[0] = pobj.first;
00544   for( unsigned int ic=0; ic<_nc; ic++ )
00545     _DOstruct[0] += DO().CTR( ic, pS, xkS, _data.ns ).first;
00546 #ifdef MC__DOSBB_DEBUG
00547   std::cout << "DO <- " << _DOstruct[0] << std::endl;
00548   int dum; std::cin >> dum;
00549 #endif
00550 
00551   // Variables to exclude from branching
00552   SBB<T>::_exclude_vars.clear();
00553   std::map<int,bool>& DOdep = _DOstruct[0].dep();
00554   for( unsigned int ip=0; ip<_np; ip++ ){
00555     std::map<int,bool>::iterator it = DOdep.find(ip);
00556     if( it==DOdep.end() || it->second ) SBB<T>::_exclude_vars.insert(ip);
00557   }
00558 
00559   // Structure of objective function and constraints (parameter & state dependence)
00560   for( unsigned int is=0, ixs=_np; is<=_data.ns; is++ )
00561     for( unsigned int ix=0; ix<_nx; ix++, ixs++)
00562       xkS[is][ix].indep(ixs);
00563   _DOstruct[1] = DO().OBJ( pS, xkS, _data.ns ).first;
00564   for( unsigned int ic=0; ic<_nc; ic++ )
00565     _DOstruct[1] += DO().CTR( ic, pS, xkS, _data.ns ).first;
00566 #ifdef MC__DOSBB_DEBUG
00567   std::cout << "DO <- " << _DOstruct[1] << std::endl;
00568   std::cin >> dum;
00569 #endif
00570 
00571   // clean up
00572   delete[] pS;
00573   for( unsigned int is=0; is<=_data.ns; is++ )
00574     delete[] xkS[is];
00575   delete[] xkS;
00576 
00577   return pb;
00578 }

00617   std::ostream&os )
00618 {
00619   // Keep track of bounds, initial guess and time stages
00620   _data.p0 = p0;
00621   _data.P  = P;
00622   _data.tk = tk;
00623   _data.ns = ns;
00624 
00625   // Keep track of execution times
00626   stats.cumul_ODEBND = stats.cumul_FPREL = stats.cumul_REDUC = stats.cumul_DOREL
00627     = stats.cumul_DOLOC = 0.;
00628 
00629   // Determine DO problem structure
00630   typename SBB<T>::TYPE pb = _set_DO_structure();
00631   _init_IVP_relax();
00632 
00633   // Run SBB solver
00634   std::set<unsigned int>&exclude = SBB<T>::_exclude_vars;
00635   SBB<T>::variables( _np, P, &exclude );
00636   stats.cumul_SBB = -time();
00637   const std::pair<double,const double*>& opt = SBB<T>::solve( pb, p0, os );
00638   stats.cumul_SBB += time();
00639   return opt;

00732   for( unsigned int is=0, ixs=_np; !failed && is<=_data.ns; is++ )
00733     for( unsigned int ix=0; ix<_nx; ix++, ixs++ ){

01014 {

01016   stats.cumul_FPREL -= time();

01165 template <typename T, typename DO>
01166 inline typename SBB<T>::STATUS
01167 DOSBB<T,DO>::_subproblem_relaxed
01168 ( const unsigned int np, T*P, double*p, double&f, const double INC )
01169 {
01170   stats.cumul_FPREL -= time();
01171   FPVT* FPVxk[_data.ns];
01172   for( unsigned int is=0; is<=_data.ns; is++ )
01173     FPVxk[is] = new FPVT[_nx];
01174   stats.cumul_FPREL += time();
01175 
01176   // Main loop for relaxation and domain reduction
01177   for( unsigned int ired=0; !ired || ired < options.DOMAIN_REDUCTION_MAXLOOPS; ired++ ){
01178 
01179     // Set current variables
01180     stats.cumul_FPREL -= time();
01181     _pFPR->reset();
01182     for( unsigned int ip=0; ip<_np; ip++ )
01183       _pFPV[ip] = FPVT( _pFPR, ip, P[ip] );
01184 #ifdef MC__DOSBB_DEBUG
01185     std::cout << "FPR:\n" << *_pFPR;
01186 #endif
01187     stats.cumul_FPREL += time();
01188 
01189     // State polyhedral relaxation
01190     typename SBB<T>::STATUS status;
01191     try{
01192       switch( options.IVP_BOUNDING_TYPE ){
01193       // State polyhedral relaxations via Intervals
01194       case Options::IA:
01195         status = _bound_IVP_IA( p, FPVxk, options.IVP_BOUNDING_PROPAGATION ); break;
01196       // State polyhedral relaxations via McCormick relaxations
01197       case Options::MC:
01198         status = _bound_IVP_MC( p, FPVxk ); break;
01199       // State polyhedral relaxations via Taylor models
01200       case Options::TM:
01201         status = _bound_IVP_TM( p, FPVxk, options.IVP_BOUNDING_PROPAGATION ); break;
01202       // State polyhedral relaxations via McCormick-Taylor models
01203       case Options::TMMC:
01204         status = _bound_IVP_TMMC( p, FPVxk ); break;
01205       default:
01206         for( unsigned int is=0; is<=_data.ns; is++ ) delete[] FPVxk[is];
01207         throw Exceptions( Exceptions::UNDEF );
01208       }
01209 
01210      // State bounding/relaxation unsuccessful
01211      if( status != SBB<T>::NORMAL ){
01212         for( unsigned int is=0; is<=_data.ns; is++ ) delete[] FPVxk[is];
01213 #ifdef MC__DOSBB_SHOW_FAILURE
01214         std::cout << "FAILURE IN ODE BOUNDING: t=" << _pODEBND->final_time()
01215                   << std::endl;
01216 #endif
01217         return SBB<T>::FAILURE;
01218       }
01219     }
01220     catch(...){
01221 #ifdef MC__DOSBB_SHOW_FAILURE
01222       std::cout << "FAILURE IN ODE BOUNDING: t=" << _pODEBND->final_time()
01223                 << std::endl;
01224 #endif
01225       return SBB<T>::FAILURE;
01226     }
01227 
01228     // Cost and constraint polyhedral relaxation
01229     std::pair<FPVT,DOSTRUCT::t_OBJ> DOobj;
01230     status = _relax_cost_constraints( FPVxk, INC, DOobj );
01231     if( status != SBB<T>::NORMAL ){
01232         for( unsigned int is=0; is<=_data.ns; is++ ) delete[] FPVxk[is];
01233         return status;
01234     }
01235 
01236     // Node processing: Domain reduction
01237     if( options.USE_DOMAIN_REDUCTION && ired < options.DOMAIN_REDUCTION_MAXLOOPS ){
01238       double maxred;
01239       stats.cumul_REDUC -= time();
01240       status = _domain_reduction( FPVxk, P, maxred );
01241       stats.cumul_REDUC += time();
01242       if( status != SBB<T>::NORMAL ){
01243         for( unsigned int is=0; is<=_data.ns; is++ ) delete[] FPVxk[is];
01244         return status;
01245       }
01246 
01247       // Reconstruct lower bounding problem when sufficient tightening was achieved
01248 #ifdef MC__DOSBB_SHOW_REDUCTION
01249       std::cerr << "max. domain reduction: " << maxred << std::endl;
01250 #endif
01251       if( maxred > options.DOMAIN_REDUCTION_THRESHOLD
01252        && ired+1 < options.DOMAIN_REDUCTION_MAXLOOPS ) continue;
01253     }
01254 
01255     // Solve relaxed problem
01256     stats.cumul_DOREL -= time();
01257     status = _relaxation_solution( DOobj, p, f );
01258     stats.cumul_DOREL += time();
01259     for( unsigned int is=0; is<=_data.ns; is++ ) delete[] FPVxk[is];
01260     return status;
01261   }
01262 
01263 #ifdef MC__DOSBB_SHOW_FAILURE
01264   std::cout << "FAILURE AT EXIT\n"; 
01265 #endif
01266   return SBB<T>::FAILURE;
01267 }
