00471 template <typename DO> inline
00472 bool
00473 DOSEQ<DO>::get_nlp_info
00474 ( Ipopt::Index& n, Ipopt::Index& m, Ipopt::Index& nnz_jac_g,
00475   Ipopt::Index& nnz_h_lag, IndexStyleEnum& index_style )
00476 {
00477 #ifdef MC__DOSEQ_TRACE
00478     std::cout << "  DOSEQ<DO>::get_nlp_info\n";
00479 #endif
00480   // set size
00481   n = _np; m = _nc;
00482 
00483   // determine IVP sparsity
00484   Structure *pS = new Structure[_np];
00485   for( unsigned int ip=0; ip<_np; ip++ )
00486     pS[ip].indep(ip);   
00487   Structure **xkS = new Structure*[_data.ns+1];
00488   xkS[0] = new Structure[_nx];
00489   for( unsigned int ix=0; ix<_nx; ix++ ){
00490     xkS[0][ix] = DO().IC( ix, pS );
00491 #ifdef MC__DOSEQ_DEBUG
00492     std::cout << "xkS[0][" << ix << "] <- " << xkS[0][ix]
00493               << std::endl;
00494 #endif
00495   }
00496   for( unsigned int is=0; is<_data.ns; is++ ){
00497     xkS[is+1] = new Structure[_nx];
00498     for( unsigned int ix=0; ix<_nx; ix++ )
00499       xkS[is+1][ix] = DO().RHS( ix, pS, xkS[is], is ) + xkS[is][ix];
00500     bool iterate = true;
00501     while( iterate ){
00502       iterate = false;
00503       for( unsigned int ix=0; ix<_nx; ix++ ){
00504         Structure xiSnew = DO().RHS( ix, pS, xkS[is+1], is ) + xkS[is+1][ix];
00505         if( xkS[is+1][ix] != xiSnew ){
00506      xkS[is+1][ix] = xiSnew; iterate = true;
00507    }
00508       }
00509     }
00510 #ifdef MC__DOSEQ_DEBUG
00511     for( unsigned int ix=0; ix<_nx; ix++ )
00512       std::cout << "xkS[" << is+1 << "][" << ix << "] <- " << xkS[is+1][ix]
00513                 << std::endl;
00514 #endif
00515   }
00516 
00517   // determine constraint sparsity
00518   delete[] _evaluation.gstruct;
00519   _evaluation.gstruct = new std::map<int,bool>[_nc];
00520   nnz_jac_g = 0;
00521   for( unsigned int ic=0; ic<_nc; ic++ ){
00522     _evaluation.gstruct[ic] = DO().CTR( ic, pS, xkS, _data.ns ).first.dep();
00523     nnz_jac_g += _evaluation.gstruct[ic].size();
00524   }
00525 
00526   // determine Lagrangian sparsity
00527   delete _evaluation.Lstruct;
00528   Structure L = DO().OBJ( pS, xkS, _data.ns ).first;
00529 #ifdef MC__DOSEQ_DEBUG
00530   std::cout << "h_lag:" << L << std::endl;
00531 #endif
00532   for( unsigned int ic=0; ic<_nc; ic++ ){
00533     L += DO().CTR( ic, pS, xkS, _data.ns ).first;
00534 #ifdef MC__DOSEQ_DEBUG
00535     std::cout << "h_lag:" << L << std::endl;
00536 #endif
00537   }
00538   _evaluation.Lstruct = new std::map<int,bool>( L.dep() ); 
00539   if( !options.SPARSE )
00540     nnz_h_lag = (n*(n+1))/2;
00541   else{
00542     int nnz_jac_lag = 0;
00543     std::map<int,bool>::const_iterator cit = _evaluation.Lstruct->begin();
00544     for( ; cit != _evaluation.Lstruct->end(); ++cit )
00545       if( !cit->second ) nnz_jac_lag++;
00546     nnz_h_lag = (nnz_jac_lag*(nnz_jac_lag+1))/2;
00547   }
00548 
00549   // use the C style indexing (0-based)
00550   index_style = Ipopt::TNLP::C_STYLE;
00551 
00552 #ifdef MC__DOSEQ_DEBUG
00553   std::cout << "n:" << n << std::endl;
00554   std::cout << "m:" << m << std::endl;
00555   std::cout << "nnz_jac_g:" << nnz_jac_g << std::endl;
00556   std::cout << "nnz_h_lag:" << nnz_h_lag << std::endl;
00557   std::cout << "h_lag:" << L << std::endl;
00558 #endif
00559 
00560   // clean up
00561   delete[] pS;
00562   for( unsigned int is=0; is<=_data.ns; is++ )
00563     delete[] xkS[is];
00564   delete[] xkS;
00565 
00566   return true;
00567 }

00587

00589   Structure *pS  = new Structure[_np];
00590   Structure **xkS = new Structure*[_data.ns+1];
00591   for( unsigned int is=0; is<=_data.ns; is++ )
00592     xkS[is] = new Structure[_nx];
00593 
00594   for( unsigned int ic=0; ic<_nc; ic++ ){
00595     switch( DO().CTR( ic, pS, xkS, _data.ns ).second ){
00596       case EQ: g_l[ic] = g_u[ic] = 0.; break;
00597       case LE: g_l[ic] = -INF; g_u[ic] = 0.; break;

00603 #endif
00604   }
00605 
00606   delete[] pS;
00607   for( unsigned int is=0; is<=_data.ns; is++ )
00608     delete[] xkS[is];
00609   delete[] xkS;
