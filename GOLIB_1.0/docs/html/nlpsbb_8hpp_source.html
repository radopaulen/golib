00452 NLPSBB<T,NLP>::solve

00454 {
00455   // Keep track of bounds and initial guess
00456   _data.p0 = p0;
00457   _data.P  = P;
00458 
00459   // Keep track of execution times
00460   stats.cumul_FPREL = stats.cumul_REDUC = stats.cumul_NLPREL
00461   = stats.cumul_NLPLOC = 0.;
00462 
00463   // Determine NLP problem structure
00464   typename SBB<T>::TYPE pb = _set_NLP_structure();
00465   _init_NLP_relax();
00466 
00467   // Run SBB solver
00468   std::set<unsigned int>&exclude = SBB<T>::_exclude_vars;
00469   SBB<T>::variables( _np, P, &exclude );
00470   stats.cumul_SBB = -time();
00471   const std::pair<double,const double*>& opt = SBB<T>::solve( pb, p0, os );
00472   stats.cumul_SBB += time();
00473   return opt;

00511 template <typename T, typename NLP>
00512 inline typename SBB<T>::STATUS
00513 NLPSBB<T,NLP>::_subproblem_feasibility
00514 ( const unsigned int np, const double*p, double&f )
00515 {
00516   // Check current point feasibility
00517   double maxinfeas = 0.;
00518   std::pair<double,NLPSTRUCT::t_CTR> g;
00519   for( unsigned int ic=0; ic<_nc; ic++ ){
00520     g = NLP().CTR( ic, p );
00521     switch( g.second ){
00522     case NLPSTRUCT::EQ:
00523       maxinfeas = std::max(maxinfeas,std::fabs(g.first)); break;
00524     case NLPSTRUCT::LE:
00525       maxinfeas = std::max(maxinfeas,g.first); break;
00526     case NLPSTRUCT::GE:
00527       maxinfeas = std::max(maxinfeas,-g.first); break;
00528     }
00529   }
00530 
00531   // Compute objective function value
00532   f = NLP().OBJ( p ).first;
00533 
00534   return( maxinfeas < _pIPOPT->options.PRIMALTOL?
00535           SBB<T>::NORMAL: SBB<T>::INFEASIBLE );
00536 }

00753               if( !Op<T>::inter(  P[ip], P[ip],
00754                 T(-SBB<T>::options.INF,_pFPR->get_objective()) ) ){
00755                 P[ip] = Op<T>::l( P[ip] );
00756                 //throw Exceptions( Exceptions::REDUC );
00757                 //return SBB<T>::INFEASIBLE;
00758               }
00759               break;
00760             case 1: // range lower bound
00761               if( !Op<T>::inter(  P[ip], P[ip],
00762                   T(_pFPR->get_objective(),SBB<T>::options.INF) ) ){
00763                 P[ip] = Op<T>::u( P[ip] );
00764                 //throw Exceptions( Exceptions::REDUC );
00765                 //return SBB<T>::INFEASIBLE;
00766               }
00767               break;
00768             }
00769             _pFPR->update_bounds( _pFPV[ip], P[ip] );
00770 #ifdef MC__NLPSBB_DEBUG
00771             std::cout << "  P(" << ip << ") = " << P[ip] << std::endl;
00772 #endif
00773             break;
00774 #ifdef USE_CPLEX
00775           case IloAlgorithm::Infeasible:
00776           case IloAlgorithm::InfeasibleOrUnbounded:
00777 #else
00778           case GRB_INFEASIBLE:
00779           case GRB_INF_OR_UNBD:
00780 #endif
00781             stats.cumul_REDUC += time();
00782             return SBB<T>::INFEASIBLE;
00783           default:
00784             stats.cumul_REDUC += time();
00785 #ifdef MC__NLPSBB_SHOW_FAILURE
00786             std::cout << "FAILURE IN DOMAIN REDUCTION LP\n"; 
00787 #endif
00788             return SBB<T>::FAILURE;
00789           }
00790         }
00791         maxred = std::max( 1.-Op<T>::diam(P[ip])/Op<T>::diam(P0), maxred );
00792       }
00793       stats.cumul_REDUC += time();
00794 
00795       // Reconstruct lower bounding problem for tightened bounds
00796 #ifdef MC__NLPSBB_SHOW_REDUCTION
00797       std::cerr << "max. domain reduction: " << maxred << std::endl;
00798 #endif
00799       if( maxred > options.DOMAIN_REDUCTION_THRESHOLD
00800        && ired+1 < options.DOMAIN_REDUCTION_MAXLOOPS ) continue;
00801     }
00802 
00803     // Compute lower bound
00804     stats.cumul_NLPREL -= time();
00805     if( pCut_inc ) _pFPR->remove_constraint( pCut_inc );
00806     switch( pobj.second ){
00807       case NLPSTRUCT::MIN: _pFPR->set_objective( FPRT::MIN, pobj.first ); break;
00808       default:       _pFPR->set_objective( FPRT::MAX, pobj.first ); break;
00809     }
00810 #ifdef MC__NLPSBB_DEBUG
00811     switch( _pFPR->solve( 1, true ) ){
00812 #else
00813     switch( _pFPR->solve() ){
00814 #endif
00815 #ifdef USE_CPLEX
00816     case IloAlgorithm::Optimal:
00817 #else
00818     case GRB_OPTIMAL:
00819 #endif
00820       // Copy solution value and solution point
00821       f = _pFPR->get_objective();
00822 #ifdef MC__NLPSBB_DEBUG
00823       std::cout << "  frel = " << f << std::endl;
00824 #endif
00825       for( unsigned int ip=0; ip<_np; ip++ ){
00826         p[ip] = _pFPR->get_variable(ip);
00827 #ifdef MC__NLPSBB_DEBUG
00828         std::cout << std::scientific << std::setprecision(4)
00829                   << "  prel(" << ip << ") = " << p[ip] << std::endl;
00830 #endif
00831       }
00832       stats.cumul_NLPREL += time();
00833       return SBB<T>::NORMAL;
00834 #ifdef USE_CPLEX
00835     case IloAlgorithm::Infeasible:
00836     case IloAlgorithm::InfeasibleOrUnbounded:
00837 #else
00838     case GRB_INFEASIBLE:
00839     case GRB_INF_OR_UNBD:
00840 #endif
00841       stats.cumul_NLPREL += time();
00842       return SBB<T>::INFEASIBLE;
00843     default:
00844 #ifdef MC__NLPSBB_SHOW_FAILURE
00845       std::cout << "FAILURE IN LOWER BOUNDING LP\n"; 
00846 #endif
00847       stats.cumul_NLPREL += time();
00848       return SBB<T>::FAILURE;
00849     }
00850   }
00851 #ifdef MC__NLPSBB_SHOW_FAILURE
00852   std::cout << "FAILURE AT EXIT\n"; 
00853 #endif
00854   stats.cumul_REDUC += time();

00866   // UPPER BOUND

00871     default:

00874 
00875   // LOWER BOUND
00876   case SBB<T>::LOWERBD:
00877     switch( SBB<T>::_pb ){
00878     case SBB<T>::MAXIMIZE:
00879       return _subproblem_local( np, P, p, f );
00880     default:
00881       return _subproblem_relaxed( np, P, p, f, INC );
00882     }
00883 
00884   // FEASIBILITY TEST
00885   case SBB<T>::FEASTEST:
00886     return _subproblem_feasibility( np, p, f );
00887 
00888   // PRE/POST-PROCESSING
00889   case SBB<T>::PREPROC:
00890   case SBB<T>::POSTPROC:
00891     return SBB<T>::NORMAL;
00892 
00893   default:
00894     return SBB<T>::FAILURE;
