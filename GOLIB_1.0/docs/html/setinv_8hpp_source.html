00425 template <typename T, typename NODE, typename LT_NODE>
00426 inline std::pair< double, double >
00427 SetInv<T,NODE,LT_NODE>::solve
00428 ( std::ostream&os )
00429 {
00430   // Create and add root node to set _open_nodes
00431   _restart();
00432   _display_init();
00433   _display( os );
00434   NODE* rootNode = new NODE( this, _P_root );
00435   _open_nodes.insert( rootNode );
00436   _open_volume = _volume_root = rootNode->volume();
00437   
00438   // Run iterative set-inversion algorithm
00439   for( _node_index = 1; !_open_nodes.empty() && _tcur-_tstart < options.MAX_CPU_TIME
00440        && ( !options.MAX_NODES || _node_index <= options.MAX_NODES );
00441        _node_index++, _tcur=time() ){
00442 
00443     // Display
00444     _display_open( os );
00445     _display_add( _node_index );
00446     _display_add( (*_open_nodes.begin())->strength() );
00447     _display_add( (unsigned int)_open_nodes.size() );
00448     _display_add( time()-_tstart );
00449     _display_add( _inner_volume );
00450     _display_add( _open_volume );
00451 
00452     // Select node and remove it for the set _open_nodes
00453     NODE* pNode = *_open_nodes.begin();
00454     _open_nodes.erase( _open_nodes.begin() );
00455     _open_volume -= pNode->volume();
00456 
00457     // Display
00458     _display_add( pNode->iter() );
00459 
00460     // Assess node
00461     _status = pNode->assess();
00462     switch( _status ){
00463     case OUTER:
00464       delete pNode;
00465       _display_add( "OUTER" );
00466       _display( os );
00467       continue;
00468 
00469     case INNER:
00470       _inner_nodes.insert( pNode );
00471       _inner_volume += pNode->volume();
00472       _display_add( "INNER" );
00473       _display( os );
00474       continue;
00475 
00476     case UNDETERMINED:
00477       // No break here - continue to branching
00478       
00479     case FAILURE:
00480       // Domain branching
00481       if( _branch_node( pNode ) == ABORT )
00482         _display_add( "ABORT" );
00483       delete pNode;
00484       _display( os );
00485       break;
00486 
00487     case ABORT: default:
00488       delete pNode;
00489       _display_add( "ABORT" );
00490       _display( os );
00491       break;
00492     }
00493 
00494     // Keep track of the maximum nodes in memory
00495     if( _open_nodes.size() > _node_max ) _node_max = _open_nodes.size();
00496 
00497     // Check status and termination criteria
00498     if( _status == ABORT || _terminate() ) break;
00499   }
00500 
00501   _display_final();
00502   _display( os );
00503   
00504   return std::make_pair( _inner_volume, _open_volume );
00505 }

00557 template <typename T, typename NODE, typename LT_NODE>
00558 inline std::pair<unsigned int, double>
00559 SetInv<T,NODE,LT_NODE>::_select_branching_variable
00560 ( NODE*pNode, const std::set<unsigned int>&set_branch )
00561 {
00562   std::pair<unsigned int, double> branchsel( _np, -1. ); // <- Can be any negative number
00563   
00564   switch( options.BRANCHING_VARIABLE_CRITERION ){
00565   // Branching based on relative range diameter
00566   case Options::RGREL:{
00567     std::set<unsigned int>::iterator it = set_branch.begin();
00568     for( ; it != set_branch.end(); ++it ){
00569       double score = Op<T>::diam( pNode->P(*it) )
00570                    / Op<T>::diam( _P_root[*it] );
00571       if( score > branchsel.second )
00572         branchsel = std::make_pair( *it, score );
00573     }
00574     break;
00575    }
00576   // Branching based on absolute range diameter
00577   case Options::RGABS:{
00578     std::set<unsigned int>::iterator it = set_branch.begin();
00579     for( ; it != set_branch.end(); ++it ){
00580       double score = Op<T>::diam( pNode->P(*it) );
00581       if( score > branchsel.second )
00582         branchsel = std::make_pair( *it, score );
00583     }
00584     break;
00585    }
00586   }
00587 
00588   return branchsel;
00589 }

00593 SetInv<T,NODE,LT_NODE>::_partition_variable_domain
00594 ( NODE*pNode, const unsigned int ip_branch )
00595 const
00596 {
00597   // Branch at mid-point of current variable range

00607 inline typename SetInv<T,NODE,LT_NODE>::STATUS
00608 SetInv<T,NODE,LT_NODE>::_branch_node
00609 ( NODE*pNode )
00610 {
00611   // Branching variable selection
00612   const unsigned int ip_branch
00613     = _select_branching_variable( pNode, _branch_set ).first;
00614   if( _status == ABORT ) return ABORT;
00615 
00616   // Partitionning
00617   std::pair<const T, const T> partition
00618     = _partition_variable_domain( pNode, ip_branch );
00619   for( unsigned int ip=0; ip<_np; ip++ ) _P_tmp[ip] = pNode->P(ip);
00620 
00621   // Left child node
00622   _P_tmp[ip_branch] = partition.first;
00623   NODE*pNodeL = new NODE( this, _P_tmp, pNode->volume(),
00624     ++_node_cnt, pNode->depth()+1, _node_index, LEFT,
00625     std::make_pair(ip_branch, pNode->P(ip_branch)), pNode->model() );
00626   _open_nodes.insert( pNodeL );
00627   _open_volume += pNodeL->volume();
00628   
00629   // Right chold node
00630   _P_tmp[ip_branch] = partition.second;
00631   NODE*pNodeR = new NODE( this, _P_tmp, pNode->volume(),
00632     ++_node_cnt, pNode->depth()+1, _node_index, RIGHT,
00633     std::make_pair(ip_branch, pNode->P(ip_branch)), pNode->model() );
00634   _open_nodes.insert( pNodeR );
00635   _open_volume += pNodeR->volume();
00636 
00637   // Display
00638   std::ostringstream omsg;
00639   omsg << "BRANCH" << ip_branch;
00640   _display_add( omsg.str() );
00641 
00642   return UNDETERMINED;
