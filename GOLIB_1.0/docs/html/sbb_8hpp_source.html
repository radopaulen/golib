00745   NODESTAT stat;
00746   switch( pNode->lower_bound( _f_inc ) ){
00747 
00748     case INFEASIBLE:
00749       if( !strongbranching ) _display_add( "INFEASIBLE" );
00750       if( relaxed ){
00751         pNode->fLB() = options.INF;
00752         pNode->update_pseudocost( _pb );
00753       }
00754       stat = FATHOM; break;
00755       
00756     case NORMAL:{
00757       if( !strongbranching ) _display_add( pNode->fLB() );
00758       if( relaxed ){
00759         // update variable pseudo-cost
00760         pNode->update_pseudocost( _pb );
00761         // fathom by value dominance?
00762         if( _fathom_by_dominance( pNode ) )
00763           { stat = FATHOM; break; }
00764         // try and update incumbent w/ relaxation solution point if applicable
00765         if( _feasible_relax( pNode )==NORMAL
00766          && _update_incumbent( pNode->fUB(), pNode->pLB() ) )
00767           { stat = UPDATED; break; }
00768       }
00769       else{
00770         // try and update incumbent
00771         if( _update_incumbent( pNode->fLB(), pNode->pLB() ) )
00772           { stat = UPDATED; break; }
00773       }
00774       stat = NOTUPDATED; break;
00775     }
00776 
00777     case FAILURE:
00778       if( !strongbranching ) _display_add( "FAILURE" );
00779       // retain node (to be sure...)
00780       pNode->fLB() = -options.INF;
00781       stat = NOTUPDATED; break;
00782 
00783     case FATAL: default:
00784       if( !strongbranching ) _display_add( "FATAL" );
00785       stat = ABORT; break;
00786   }

00895 template <typename T>

00899 {
00900   std::pair<unsigned int, double> branchsel( _np, -options.INF );
00901   
00902   switch( options.BRANCHING_VARIABLE_CRITERION ){
00903   // Branching based on pseudocosts
00904   case Options::RELIAB:
00905     if( _reliable_pseudocost( pNode, set_branch ) ){
00906       std::set<unsigned int>::iterator it = set_branch.begin();
00907 #ifdef DEBUG__SBB_BRANCHINGVAR
00908       std::cout << "*** SCORES";
00909 #endif
00910       for( ; it != set_branch.end(); ++it ){
00911         std::pair<const T, const T> partition
00912           = _partition_variable_domain( pNode, *it );
00913 #ifdef USE__SBB_PSEUDOCOST_FILTER

00915           * pNode->pc_var()[*it].second, Op<T>::diam( partition.first )
00916           * pNode->pc_var()[*it+_np].second );
00917 #else
00918         double score = _score_branching( Op<T>::diam( partition.second )
00919           * pNode->pc_var()[*it].second / pNode->pc_var()[*it].first,
00920                                          Op<T>::diam( partition.first )
00921           * pNode->pc_var()[*it+_np].second / pNode->pc_var()[*it+_np].first );
00922 #endif
00923 #ifdef DEBUG__SBB_BRANCHINGVAR
00924         std::cout << "   " << *it << ": " << score;
00925 #endif
00926         if( score > branchsel.second ) branchsel = std::make_pair( *it, score );
00927       }
00928 #ifdef DEBUG__SBB_BRANCHINGVAR
00929       std::cout << std::endl;
00930 #endif
00931       break;
00932     }
00933     // Unreliable pseudocosts - Use strong branching
00934     branchsel = _strong_branching( pNode, set_branch );
00935     if( _REL_stat == ABORT || branchsel.first < _np ) break;
00936     // No break - use RGREL criterion in case strong branching failed
00937 
00938   // Branching based on relative range diameter
00939   case Options::RGREL:{
00940     std::set<unsigned int>::iterator it = set_branch.begin();
00941     for( ; it != set_branch.end(); ++it ){
00942       double score = Op<T>::diam( pNode->P(*it) )
00943                    / Op<T>::diam( _P_root[*it] );
00944       if( score > branchsel.second )
00945         branchsel = std::make_pair( *it, score );
00946     }
00947     break;
00948    }
00949   // Branching based on absolute range diameter
00950   case Options::RGABS:{
00951     std::set<unsigned int>::iterator it = set_branch.begin();
00952     for( ; it != set_branch.end(); ++it ){
00953       double score = Op<T>::diam( pNode->P(*it) );
00954       if( score > branchsel.second )
00955         branchsel = std::make_pair( *it, score );
00956     }
00957     break;
00958    }
00959   }

00979 template <typename T>
00980 inline std::pair<const T, const T>
00981 SBB<T>::_partition_variable_domain
00982 ( SBBNode<T>*pNode, const unsigned int ip_branch )
00983 const
00984 {
00985   std::pair<T,T> partition;
00986 
00987   switch( options.BRANCHING_STRATEGY ){
00988   case Options::OMEGA:
00989     // Branch at incumbent if interior to current variable range
00990     if( _p_inc && !pNode->at_bound(_p_inc[ip_branch], ip_branch,
00991       options.BRANCHING_BOUND_THRESHOLD) ){
00992       partition.first = mc::Op<T>::l(pNode->P(ip_branch)) + Op<T>::zeroone()
00993         *( _p_inc[ip_branch] - mc::Op<T>::l(pNode->P(ip_branch)) );
00994       partition.second = mc::Op<T>::u(pNode->P(ip_branch)) + Op<T>::zeroone()
00995         *( _p_inc[ip_branch] - mc::Op<T>::u(pNode->P(ip_branch)) );
00996       break;
00997     }
00998     // Otherwise branch at relaxation solution point if available +
00999     // interior to current variable range
01000     if( _pb == MINIMIZE && pNode->fLB() > -options.INF
01001      && !pNode->at_bound(pNode->pLB(ip_branch), ip_branch,
01002      options.BRANCHING_BOUND_THRESHOLD) ){ 
01003       partition.first = mc::Op<T>::l(pNode->P(ip_branch)) + Op<T>::zeroone()
01004         *( pNode->pLB(ip_branch) - mc::Op<T>::l(pNode->P(ip_branch)) );
01005       partition.second = mc::Op<T>::u(pNode->P(ip_branch)) + Op<T>::zeroone()
01006         *( pNode->pLB(ip_branch) - mc::Op<T>::u(pNode->P(ip_branch)) );
01007       break;
01008     }
01009     else if( _pb == MAXIMIZE && pNode->fUB() < options.INF
01010      && !pNode->at_bound(pNode->pUB(ip_branch), ip_branch,
01011      options.BRANCHING_BOUND_THRESHOLD) ){ 
01012       partition.first = mc::Op<T>::l(pNode->P(ip_branch)) + Op<T>::zeroone()
01013         *( pNode->pUB(ip_branch) - mc::Op<T>::l(pNode->P(ip_branch)) );
01014       partition.second = mc::Op<T>::u(pNode->P(ip_branch)) + Op<T>::zeroone()
01015         *( pNode->pUB(ip_branch) - mc::Op<T>::u(pNode->P(ip_branch)) );
01016       break;
01017     }
01018     
01019   case Options::MIDPOINT: default:
01020     // Branch at mid-point of current variable range
01021     partition.first = mc::Op<T>::l(pNode->P(ip_branch)) + Op<T>::zeroone()
01022       *Op<T>::diam(pNode->P(ip_branch))/2.;
01023     partition.second = mc::Op<T>::u(pNode->P(ip_branch)) - Op<T>::zeroone()
01024       *Op<T>::diam(pNode->P(ip_branch))/2.;
01025     break;
01026   }
01027 
01028   return partition;
01029 }

01085 template <typename T>
01086 inline std::pair<unsigned int, double>
01087 SBB<T>::_strong_branching
01088 ( SBBNode<T>*pNode, const std::set<unsigned int>&set_branch )
01089 {
01090   std::pair<unsigned int, double> branchsel( _np, -options.INF );
01091   if( !options.BRANCHING_RELIABILITY_THRESHOLD ) return branchsel;
01092 
01093   // Create child subnode
01094   //for( unsigned int ip=0; ip<_np; ip++ ) _P_tmp[ip] = pNode->P(ip);
01095   SBBNode<T>* pSubnode = 0;
01096   switch( _pb ){
01097   case MINIMIZE:
01098     if( pNode->fLB() == -options.INF ) return branchsel;
01099     pSubnode = new SBBNode<T>( this, pNode->P(), pNode->fLB(), pNode->index(),
01100       pNode->depth()+1, _node_index, LEFT, std::make_pair(0,pNode->P(0)), pNode->pc_var() ); break;
01101   case MAXIMIZE:
01102     if( pNode->fUB() == options.INF ) return branchsel;
01103     pSubnode = new SBBNode<T>( this, pNode->P(), -pNode->fUB(), pNode->index(),
01104       pNode->depth()+1, _node_index, LEFT, std::make_pair(0,pNode->P(0)), pNode->pc_var() ); break;
01105   }
01106   
01107   // Repeat for all variables in branching set
01108 #ifdef DEBUG__SBB_STRONGBRANCHING
01109   std::cout << "*** SCORES";
01110 #endif
01111   double fLEFT, fRIGHT, fSCORE;
01112   std::set<unsigned int>::iterator it = set_branch.begin();
01113   for( ; it != set_branch.end(); ++it ){
01114 
01115     // Simulate partitionning
01116     pSubnode->parent() = std::make_pair( *it, pNode->P(*it) );
01117     const std::pair<const T, const T> partition
01118       = _partition_variable_domain( pNode, *it );
01119 
01120     // Relax left partition
01121     pSubnode->P(*it) = partition.first;
01122     pSubnode->type() = LEFT;
01123     switch( _pb ){
01124     case MINIMIZE:
01125       _REL_stat = _lower_bound( pSubnode, true, true );
01126       fLEFT = pSubnode->fLB(); break;
01127     case MAXIMIZE: default:
01128       _REL_stat = _upper_bound( pSubnode, true, true );
01129       fLEFT = -pSubnode->fUB(); break;
01130     }
01131     if( _REL_stat == ABORT ) break;
01132     // Left partition failed
01133     //if( fLEFT == -options.INF ) continue;
01134  
01135     // Relax right partition
01136     pSubnode->P(*it) = partition.second;
01137     pSubnode->type() = RIGHT;
01138     switch( _pb ){
01139     case MINIMIZE:
01140       _REL_stat = _lower_bound( pSubnode, true, true );
01141       fRIGHT = pSubnode->fLB(); break;
01142     case MAXIMIZE: default:
01143       _REL_stat = _upper_bound( pSubnode, true, true );
01144       fRIGHT = -pSubnode->fUB(); break;
01145     }
01146     if( _REL_stat == ABORT ) break;
01147     // Right partition failed
01148     //if( fRIGHT == -options.INF ) continue;
01149 
01150     // Score current variable and compare
01151     fSCORE = _score_branching( fLEFT, fRIGHT );
01152 #ifdef DEBUG__SBB_STRONGBRANCHING
01153     std::cout << "   " << *it << ": " << fSCORE
01154               << " (" << fLEFT << "," << fRIGHT << ")";
01155 #endif
01156     if( fSCORE > branchsel.second )
01157       branchsel = std::make_pair( *it, fSCORE );
01158 
01159     // Reset variable bounds
01160     for( unsigned int ip=0; ip<_np; ip++ )
01161       pSubnode->P(ip) = pNode->P(ip);
01162   }
01163 #ifdef DEBUG__SBB_STRONGBRANCHING
01164   std::cout << std::endl;
01165 #endif
01166 
01167   // Update variable pseudocosts
01168   for( unsigned int ip=0; ip<2*_np; ip++ )
01169     pNode->pc_var(ip) = pSubnode->pc_var()[ip];
01170   delete pSubnode;
01171   pNode->strongbranch();
01172 
01173   return branchsel;
01174 }
