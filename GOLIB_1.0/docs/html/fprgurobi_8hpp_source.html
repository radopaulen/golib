00249 template <typename T> inline void
00250 FPRGurobi<T>::_append_cut
00251 ( const FPCut<T>* pCut )
00252 {
00253   GRBVar* VarSOS = 0;
00254   double* WeiSOS = 0;
00255   int TypSOS = 0;
00256   if( pCut->type() == FPCut<T>::SOS1 || pCut->type() == FPCut<T>::SOS2 ){
00257     VarSOS = new GRBVar[pCut->nvar()];
00258     WeiSOS = new double[pCut->nvar()];
00259     TypSOS = ( pCut->type() == FPCut<T>::SOS1? GRB_SOS_TYPE1: GRB_SOS_TYPE2 );
00260   }
00261 
00262   GRBLinExpr lhs;
00263   for( unsigned int k=0; k<pCut->nvar(); k++ ){
00264     switch( pCut->idvar()[k].first ){
00265       case FPVar<T>::VARCONT: case FPVar<T>::VARBIN:
00266       { it_GRBVar ivar = _var.find( pCut->idvar()[k].second );
00267         if( ivar==_var.end() )
00268           throw typename FPRelax<T>::Exceptions( FPRelax<T>::Exceptions::INTERNAL );
00269         GRBVar&Var = (*ivar).second;
00270         lhs += GRBLinExpr( Var, pCut->coef()[k] );
00271         if( pCut->type() == FPCut<T>::SOS1 || pCut->type() == FPCut<T>::SOS2 ){
00272           VarSOS[k] = Var; WeiSOS[k] = (double)k/(double)pCut->nvar();
00273         }
00274         break; }
00275       default:
00276       { it_GRBVar iaux = _aux.find( pCut->idvar()[k].second );
00277         if( iaux==_aux.end() )
00278           throw typename FPRelax<T>::Exceptions( FPRelax<T>::Exceptions::INTERNAL );
00279         GRBVar&Aux = (*iaux).second;
00280         lhs += GRBLinExpr( Aux, pCut->coef()[k] );
00281         if( pCut->type() == FPCut<T>::SOS1 || pCut->type() == FPCut<T>::SOS2 ){
00282           VarSOS[k] = Aux; WeiSOS[k] = (double)k/(double)pCut->nvar();
00283         }
00284         break; }
00285     }
00286   }
00287 
00288   switch( pCut->type() ){
00289     case FPCut<T>::SOS1:
00290     case FPCut<T>::SOS2:
00291       _model->addSOS( VarSOS, WeiSOS, pCut->nvar(), TypSOS );
00292       delete [] VarSOS;
00293       delete [] WeiSOS;
00294     case FPCut<T>::EQ:
00295       _cut.insert( std::make_pair( pCut->op(), _model->addConstr( lhs,
00296         GRB_EQUAL, pCut->rhs() ) ) );
00297       break;
00298     case FPCut<T>::LE:
00299       _cut.insert( std::make_pair( pCut->op(), _model->addConstr( lhs,
00300         GRB_LESS_EQUAL, pCut->rhs() ) ) );
00301       break;
00302     case FPCut<T>::GE:
00303       _cut.insert( std::make_pair( pCut->op(), _model->addConstr( lhs,
00304         GRB_GREATER_EQUAL, pCut->rhs() ) ) );
00305       break;
00306   }
00307 }
