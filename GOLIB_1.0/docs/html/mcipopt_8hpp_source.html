00414 template <typename NLP> inline
00415 bool
00416 IPOPT<NLP>::get_nlp_info
00417 ( Ipopt::Index& n, Ipopt::Index& m, Ipopt::Index& nnz_jac_g,
00418   Ipopt::Index& nnz_h_lag, IndexStyleEnum& index_style )
00419 {
00420 #ifdef MC__IPOPT_TRACE
00421     std::cout << "  IPOPT<NLP>::get_nlp_info\n";
00422 #endif
00423   // set size
00424   n = _np; m = _nc;
00425 
00426   // determine constraint sparsity
00427   delete[] _evaluation.gstruct;
00428   _evaluation.gstruct = new std::map<int,bool>[_nc];
00429   Structure *pS = new Structure[_np];
00430   for( unsigned int ip=0; ip<_np; ip++ ) pS[ip].indep(ip);   
00431   nnz_jac_g = 0;
00432   for( unsigned int ic=0; ic<_nc; ic++ ){
00433     std::pair<Structure,t_CTR> g = NLP().CTR( ic, pS );
00434     _evaluation.gstruct[ic] = g.first.dep();
00435     nnz_jac_g += _evaluation.gstruct[ic].size();
00436   }
00437 
00438   // determine Lagrangian sparsity
00439   delete _evaluation.Lstruct;
00440   std::pair<Structure,t_OBJ> f = NLP().OBJ( pS );
00441   Structure L = f.first;
00442 #ifdef MC__IPOPT_DEBUG
00443   std::cout << "h_lag:" << L << std::endl;
00444 #endif
00445   for( unsigned int ic=0; ic<_nc; ic++ ){
00446     std::pair<Structure,t_CTR> g = NLP().CTR( ic, pS );
00447     L += g.first;
00448 #ifdef MC__IPOPT_DEBUG
00449     std::cout << "h_lag:" << L << std::endl;
00450 #endif
00451   }
00452   _evaluation.Lstruct = new std::map<int,bool>( L.dep() ); 
00453   if( !options.SPARSE )
00454     nnz_h_lag = (n*(n+1))/2;
00455   else{
00456     int nnz_jac_lag = 0;
00457     std::map<int,bool>::const_iterator cit = _evaluation.Lstruct->begin();
00458     for( ; cit != _evaluation.Lstruct->end(); ++cit )
00459       if( !cit->second ) nnz_jac_lag++;
00460     nnz_h_lag = (nnz_jac_lag*(nnz_jac_lag+1))/2;
00461   }
00462 
00463   // use the C style indexing (0-based)
00464   index_style = Ipopt::TNLP::C_STYLE;
00465 
00466 #ifdef MC__IPOPT_DEBUG
00467   std::cout << "n:" << n << std::endl;
00468   std::cout << "m:" << m << std::endl;
00469   std::cout << "nnz_jac_g:" << nnz_jac_g << std::endl;
00470   std::cout << "nnz_h_lag:" << nnz_h_lag << std::endl;
00471   std::cout << "h_lag:" << L << std::endl;
00472 #endif
00473 
00474   // clean up
00475   delete[] pS;
00476 
00477   return true;
